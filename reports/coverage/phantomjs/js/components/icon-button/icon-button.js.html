(function (app) {
 
  var moduleName = 'iconButton';
 
  try {
    app = angular.module(moduleName);

  }
 
  function iconButtonDirective() {
 
    return {
      restrict: 'E',
      templateUrl: '/js/components/icon-button/icon-button.html',
      transclude: true,

        href: '@',

  }
